\section{零七}
\vspace*{2\ccwd}
\begin{adjustwidth}{2cm}{1cm}
    \Large\kaishu{
    某年，9月9日，周二

    \vspace*{2\ccwd} 

    流过青春

    才渐晓觉

    手抓不住的

    眼睁睁
    
    目望她离我去

    痛惜，悔彻，思定

    隐怀愿许

    \vspace*{2\ccwd} 

    昨夜雾浓

    今夕月明

    旧日过往

    几多遗憾

    愿能从兹

    不为势屈

    慰告己怀

    勿违光阴

    勿愧伊人
}
\end{adjustwidth}
